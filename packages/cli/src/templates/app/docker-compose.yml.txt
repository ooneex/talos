services:
  # PostgreSQL - Primary relational database
  # Docs: https://www.postgresql.org/docs/
  # Connection: postgresql://talos:talos@localhost:5432/talos
  postgres:
    image: postgres:18.3-alpine3.23
    container_name: {{NAME}}_db
    restart: "on-failure"
    ports:
      - "5432:5432"
    volumes:
      - {{NAME}}_db_data:/var/lib/postgresql/data:rw
    environment:
      - POSTGRES_ROOT_PASSWORD=talos
      - POSTGRES_DATABASE=talos
      - POSTGRES_USER=talos
      - POSTGRES_PASSWORD=talos
      - PGDATA=/var/lib/postgresql/data/pgdata

  # Redis - In-memory data store for caching and sessions
  # Docs: https://redis.io/docs/
  # Connection: redis://localhost:6379
  redis:
    image: redis:8.4-alpine
    container_name: {{NAME}}_redis
    restart: "on-failure"
    ports:
      - "6379:6379"
    volumes:
      - {{NAME}}_redis_data:/data:rw

volumes:
  {{NAME}}_db_data:
  {{NAME}}_redis_data:
