# Bitbucket Pipelines — {{name}} microservice
#
# Bitbucket supports a single bitbucket-pipelines.yml per repository, so these
# steps cannot be auto-merged. Copy the definitions below into your root
# bitbucket-pipelines.yml (under `definitions.steps`) and reference the anchors
# from the relevant `pipelines` sections.
#
# Builds and deploys the {{name}} microservice image from modules/{{name}}/Dockerfile.
# Repo-wide lint/test run in the app's pipeline, so they are not repeated here.

definitions:
  steps:
    # The `docker` service exposes the Docker CLI to the step, so `talos
    # docker:publish` can build and push from the Bun image.
    - step: &{{name}}-build-image
        name: Build {{name}} Image
        services:
          - docker
        script:
          - bun add -g @talosjs/cli
          - SHORT_SHA=$(echo "$BITBUCKET_COMMIT" | cut -c1-7)
          - >-
            talos docker:credentials:create
            --registry "${DOCKER_REGISTRY:-docker.io}"
            --username "$DOCKER_USERNAME"
            --token "$DOCKER_TOKEN"
            --silent
          # Builds and pushes only the {{name}} microservice image from
          # modules/{{name}}/Dockerfile, tagged with the short commit SHA.
          - talos docker:publish --modules {{name}} --tag "sha-$SHORT_SHA" --silent

    - step: &{{name}}-deploy-production
        name: Deploy {{name}} to Production
        deployment: production-{{name}}
        services:
          - docker
        script:
          - bun add -g @talosjs/cli
          # Apply migrations before publishing the release image, so the database
          # is ready by the time the platform pulls the new tag.
          - APP_ENV=production talos migration:up
          - >-
            talos docker:credentials:create
            --registry "${DOCKER_REGISTRY:-docker.io}"
            --username "$DOCKER_USERNAME"
            --token "$DOCKER_TOKEN"
            --silent
          # Builds and pushes only the {{name}} microservice image from
          # modules/{{name}}/Dockerfile, tagged with the release version (e.g. v1.2.3).
          - talos docker:publish --modules {{name}} --tag "${BITBUCKET_TAG:-$BITBUCKET_BRANCH}" --silent

# Add to your `pipelines` section, e.g.:

#         - step:
