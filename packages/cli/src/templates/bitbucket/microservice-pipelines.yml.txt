# Bitbucket Pipelines — {{name}} microservice
#
# Bitbucket supports a single bitbucket-pipelines.yml per repository, so these
# steps cannot be auto-merged. Copy the definitions below into your root
# bitbucket-pipelines.yml (under `definitions.steps`) and reference the anchors
# from the relevant `pipelines` sections.
#
# Builds and deploys the {{name}} microservice image from modules/{{name}}/Dockerfile.
#
# Required repository variable (in addition to the app's):
#   PRODUCTION_URL_{{NAME_UPPER}}  — {{name}} URL used for health checks

definitions:
  steps:
    - step: &{{name}}-build-image
        name: Build {{name}} Image
        image: docker:27
        services:
          - docker
        caches:
          - docker
        script:
          - SHORT_SHA=$(echo $BITBUCKET_COMMIT | cut -c1-7)
          - IMAGE=$DOCKER_REGISTRY/{{name}}
          - echo "$DOCKER_PASSWORD" | docker login $DOCKER_REGISTRY -u $DOCKER_USER --password-stdin
          - docker buildx create --use --bootstrap
          - |
            docker buildx build \
              --cache-from type=registry,ref=$IMAGE:cache \
              --cache-to  type=registry,ref=$IMAGE:cache,mode=max \
              --tag "$IMAGE:sha-$SHORT_SHA" \
              --tag "$IMAGE:$BITBUCKET_BRANCH" \
              --build-arg APP_VERSION=$BITBUCKET_COMMIT \
              --push \
              -f modules/{{name}}/Dockerfile \
              .

    - step: &{{name}}-deploy-production
        name: Deploy {{name}} to Production
        deployment: production-{{name}}
        image: alpine:latest
        script:
          - apk add --no-cache openssh-client curl
          - mkdir -p ~/.ssh && chmod 700 ~/.ssh
          - echo "$PROD_SSH_KEY" | base64 -d > ~/.ssh/id_rsa && chmod 600 ~/.ssh/id_rsa
          - ssh-keyscan "$PROD_HOST" >> ~/.ssh/known_hosts
          - SHORT_SHA=$(echo $BITBUCKET_COMMIT | cut -c1-7)
          - IMAGE=$DOCKER_REGISTRY/{{name}}:sha-$SHORT_SHA
          - |
            ssh $PROD_USER@$PROD_HOST \
              DOCKER_REGISTRY="$DOCKER_REGISTRY" \
              DOCKER_USER="$DOCKER_USER" \
              DOCKER_PASSWORD="$DOCKER_PASSWORD" \
              IMAGE="$IMAGE" \
              'bash -s' << 'ENDSSH'
                set -e

                echo "$DOCKER_PASSWORD" | docker login "$DOCKER_REGISTRY" -u "$DOCKER_USER" --password-stdin

                # Preserve current image as rollback target before pulling new one
                docker tag "$DOCKER_REGISTRY/{{name}}:latest" "$DOCKER_REGISTRY/{{name}}:rollback" 2>/dev/null || true

                docker pull "$IMAGE"

                cd /opt/{{NAME}}

                # Run migrations before switching traffic to catch failures early
                docker run --rm --env-file .env "$IMAGE" bun run db:migrate

                APP_IMAGE="$IMAGE" docker compose up -d --no-deps --wait app
                docker image prune -f --filter "until=24h"
            ENDSSH
          - |
            echo "Verifying {{name}} health at $PRODUCTION_URL_{{NAME_UPPER}}..."
            for i in $(seq 1 24); do
              STATUS=$(curl -sf -o /dev/null -w "%{http_code}" "$PRODUCTION_URL_{{NAME_UPPER}}/health" 2>/dev/null || echo "000")
              [ "$STATUS" = "200" ] && echo "{{name}} is healthy (attempt $i)" && exit 0
              echo "Attempt $i/24 — HTTP $STATUS, retrying in 5s..."
              sleep 5
            done
            echo "Health check failed — rolling back to previous image"
            ssh $PROD_USER@$PROD_HOST \
              DOCKER_REGISTRY="$DOCKER_REGISTRY" \
              'bash -s' << 'ENDSSH'
                cd /opt/{{NAME}}
                APP_IMAGE="$DOCKER_REGISTRY/{{name}}:rollback" docker compose up -d --no-deps app
            ENDSSH
            exit 1
