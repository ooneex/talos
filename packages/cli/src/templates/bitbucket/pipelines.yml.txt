# Bitbucket Pipelines
# Place at bitbucket-pipelines.yml in the repository root.
#
# Required repository variables (Repository settings → Pipelines → Repository variables):
#   DOCKER_REGISTRY    — Container registry host and namespace
#                        Docker Hub: docker.io/myorg
#                        AWS ECR:    123456789.dkr.ecr.us-east-1.amazonaws.com
#   DOCKER_USER        — Registry username (or AWS_ACCESS_KEY_ID for ECR)
#   DOCKER_PASSWORD    — Registry password / token (or AWS_SECRET_ACCESS_KEY for ECR)
#   PROD_HOST          — Production server hostname or IP
#   PROD_USER          — SSH username
#   PROD_SSH_KEY       — SSH private key (base64-encoded: base64 -w0 ~/.ssh/id_rsa)
#   PRODUCTION_URL     — App URL used for health checks (e.g. https://app.example.com)

image: oven/bun:latest

definitions:
  caches:
    bun: ~/.bun/install/cache
    nx: .nx/cache

  steps:
    - step: &quality
        name: Quality
        caches:
          - bun
          - nx
        script:
          - bun install --frozen-lockfile
          - bun run lint
          - bun test

    - step: &build-image
        name: Build Image
        image: docker:27
        services:
          - docker
        caches:
          - docker
        script:
          - SHORT_SHA=$(echo $BITBUCKET_COMMIT | cut -c1-7)
          - IMAGE_SHA=$DOCKER_REGISTRY/{{NAME}}:sha-$SHORT_SHA
          - IMAGE_REF=$DOCKER_REGISTRY/{{NAME}}:$BITBUCKET_BRANCH
          - echo "$DOCKER_PASSWORD" | docker login $DOCKER_REGISTRY -u $DOCKER_USER --password-stdin
          - docker buildx create --use --bootstrap
          - |
            docker buildx build \
              --cache-from type=registry,ref=$DOCKER_REGISTRY/{{NAME}}:cache \
              --cache-to  type=registry,ref=$DOCKER_REGISTRY/{{NAME}}:cache,mode=max \
              --tag "$IMAGE_SHA" \
              --tag "$IMAGE_REF" \
              --build-arg APP_VERSION=$BITBUCKET_COMMIT \
              --push \
              .

    - step: &build-image-latest
        name: Build Image (latest)
        image: docker:27
        services:
          - docker
        caches:
          - docker
        script:
          - SHORT_SHA=$(echo $BITBUCKET_COMMIT | cut -c1-7)
          - IMAGE_SHA=$DOCKER_REGISTRY/{{NAME}}:sha-$SHORT_SHA
          - IMAGE_TAG=$DOCKER_REGISTRY/{{NAME}}:$BITBUCKET_TAG
          - echo "$DOCKER_PASSWORD" | docker login $DOCKER_REGISTRY -u $DOCKER_USER --password-stdin
          - docker buildx create --use --bootstrap
          - |
            docker buildx build \
              --cache-from type=registry,ref=$DOCKER_REGISTRY/{{NAME}}:cache \
              --cache-to  type=registry,ref=$DOCKER_REGISTRY/{{NAME}}:cache,mode=max \
              --tag "$IMAGE_SHA" \
              --tag "$IMAGE_TAG" \
              --tag "$DOCKER_REGISTRY/{{NAME}}:latest" \
              --build-arg APP_VERSION=$BITBUCKET_COMMIT \
              --push \
              .

    - step: &deploy-production
        name: Deploy to Production
        deployment: production
        image: alpine:latest
        script:
          - apk add --no-cache openssh-client curl
          - mkdir -p ~/.ssh && chmod 700 ~/.ssh
          - echo "$PROD_SSH_KEY" | base64 -d > ~/.ssh/id_rsa && chmod 600 ~/.ssh/id_rsa
          - ssh-keyscan "$PROD_HOST" >> ~/.ssh/known_hosts
          - SHORT_SHA=$(echo $BITBUCKET_COMMIT | cut -c1-7)
          - IMAGE=$DOCKER_REGISTRY/{{NAME}}:sha-$SHORT_SHA
          - |
            ssh $PROD_USER@$PROD_HOST \
              DOCKER_REGISTRY="$DOCKER_REGISTRY" \
              DOCKER_USER="$DOCKER_USER" \
              DOCKER_PASSWORD="$DOCKER_PASSWORD" \
              IMAGE="$IMAGE" \
              'bash -s' << 'ENDSSH'
                set -e

                echo "$DOCKER_PASSWORD" | docker login "$DOCKER_REGISTRY" -u "$DOCKER_USER" --password-stdin

                # Preserve current image as rollback target before pulling new one
                docker tag "$DOCKER_REGISTRY/{{NAME}}:latest" "$DOCKER_REGISTRY/{{NAME}}:rollback" 2>/dev/null || true

                docker pull "$IMAGE"

                cd /opt/{{NAME}}

                # Run migrations before switching traffic to catch failures early
                docker run --rm --env-file .env "$IMAGE" bun run db:migrate

                APP_IMAGE="$IMAGE" docker compose up -d --no-deps --wait app
                docker image prune -f --filter "until=24h"
            ENDSSH
          - |
            echo "Verifying production health at $PRODUCTION_URL..."
            for i in $(seq 1 24); do
              STATUS=$(curl -sf -o /dev/null -w "%{http_code}" "$PRODUCTION_URL/health" 2>/dev/null || echo "000")
              [ "$STATUS" = "200" ] && echo "Production is healthy (attempt $i)" && exit 0
              echo "Attempt $i/24 — HTTP $STATUS, retrying in 5s..."
              sleep 5
            done
            echo "Health check failed — rolling back to previous image"
            ssh $PROD_USER@$PROD_HOST \
              DOCKER_REGISTRY="$DOCKER_REGISTRY" \
              'bash -s' << 'ENDSSH'
                cd /opt/{{NAME}}
                APP_IMAGE="$DOCKER_REGISTRY/{{NAME}}:rollback" docker compose up -d --no-deps app
            ENDSSH
            exit 1

pipelines:
  # Quality gate on every pull request
  pull-requests:
    "**":
      - step: *quality

  branches:
    # Main branch: CI then manual production deploy
    main:
      - step: *quality
      - step: *build-image
      - step:
          <<: *deploy-production
          trigger: manual

    # Develop branch: CI and build only
    develop:
      - step: *quality
      - step: *build-image

  # Semver tags: CI, build with latest tag, auto-deploy to production
  tags:
    "v*.*.*":
      - step: *quality
      - step: *build-image-latest
      - step: *deploy-production
