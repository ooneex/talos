# Bitbucket Pipelines
# Place at bitbucket-pipelines.yml in the repository root.
#
# Required repository variables (Repository settings → Pipelines → Repository variables):
#   DOCKER_REGISTRY  — Container registry host (defaults to docker.io)
#   DOCKER_USERNAME  — Registry username
#   DOCKER_TOKEN     — Registry access token (mark as "Secured")
#   (plus the production database connection consumed by `talos migration:up`)

image: oven/bun:latest

definitions:
  caches:
    bun: ~/.bun/install/cache
    monorepo: var/cache/monorepo

  steps:
    - step: &quality
        name: Quality
        caches:
          - bun
          - monorepo
        script:
          - bun add -g @talosjs/cli
          - talos monorepo:check

    # The `docker` service exposes the Docker CLI to the step, so `talos
    # docker:publish` can build and push from the Bun image.
    - step: &build-image
        name: Build Image
        services:
          - docker
        script:
          - bun add -g @talosjs/cli
          - SHORT_SHA=$(echo "$BITBUCKET_COMMIT" | cut -c1-7)
          - >-
            talos docker:credentials:create
            --registry "${DOCKER_REGISTRY:-docker.io}"
            --username "$DOCKER_USERNAME"
            --token "$DOCKER_TOKEN"
            --silent
          # Builds and pushes an image for every package/module that ships a
          # Dockerfile, tagging each with the short commit SHA.
          - talos docker:publish --tag "sha-$SHORT_SHA" --silent

    - step: &deploy-production
        name: Deploy to Production
        deployment: production
        services:
          - docker
        script:
          - bun add -g @talosjs/cli
          # Apply migrations before publishing the release image, so the database
          # is ready by the time the platform pulls the new tag. Expose the
          # production database connection to this step via repository variables.
          - APP_ENV=production talos migration:up
          - >-
            talos docker:credentials:create
            --registry "${DOCKER_REGISTRY:-docker.io}"
            --username "$DOCKER_USERNAME"
            --token "$DOCKER_TOKEN"
            --silent
          # Builds and pushes an image for every package/module that ships a
          # Dockerfile, tagged with the release version (e.g. v1.2.3). Deployment
          # is handled by the registry — the platform pulls the new tag.
          - talos docker:publish --tag "${BITBUCKET_TAG:-$BITBUCKET_BRANCH}" --silent

pipelines:
  # Quality gate on every pull request
  pull-requests:
    "**":
      - step: *quality

  branches:
    # Main branch: CI then manual production deploy
    main:
      - step: *quality
      - step: *build-image
      - step:
          <<: *deploy-production
          trigger: manual

    # Staging and develop branches: CI and build only
    staging:
      - step: *quality
      - step: *build-image
    develop:
      - step: *quality
      - step: *build-image

  # Semver tags: quality gate, then auto-deploy to production
  tags:
    "v*.*.*":
      - step: *quality
      - step: *deploy-production
