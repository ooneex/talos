# Fish completion for the Talos CLI (oo / talos)
#
# https://fishshell.com/docs/current/completions.html
#
# Command names use a "group:action" form (e.g. app:create). Fish keeps the
# colon inside a single token, so __fish_seen_subcommand_from matches them as-is.

function __talos_modules
    test -d modules; or return
    command ls -1 modules 2>/dev/null
end

function __talos_packages
    test -d packages; or return
    for f in packages/*/package.json
        test -f "$f"; and string replace -r '^packages/([^/]+)/package\.json$' '$1' -- "$f"
    end
end

function __talos_pkg_modules
    test -d modules; or return
    for f in modules/*/package.json
        test -f "$f"; and string replace -r '^modules/([^/]+)/package\.json$' '$1' -- "$f"
    end
end

function __talos_docker_packages
    test -d packages; or return
    for f in packages/*/Dockerfile
        test -f "$f"; and string replace -r '^packages/([^/]+)/Dockerfile$' '$1' -- "$f"
    end
end

function __talos_docker_modules
    test -d modules; or return
    for f in modules/*/Dockerfile
        test -f "$f"; and string replace -r '^modules/([^/]+)/Dockerfile$' '$1' -- "$f"
    end
end

function __talos_yaml_modules --argument-names type
    test -d modules; or return
    # `set` tolerates a non-matching glob (unlike passing it straight to grep).
    set -l files modules/*/*.yml
    test (count $files) -gt 0; or return
    command grep -l "type: \"$type\"" $files 2>/dev/null \
        | command sed -n 's|modules/\([^/]*\)/.*|\1|p' | command sort -u
end

function __talos_design_modules
    __talos_yaml_modules design
end

function __talos_api_modules
    __talos_yaml_modules api
end

function __talos_microservice_modules
    __talos_yaml_modules microservice
end

function __talos_spa_modules
    __talos_yaml_modules spa
end

function __talos_sdk_names
    __talos_yaml_modules sdk
end

function __talos_destination_modules
    test -d modules; or return
    set -l files modules/*/*.yml
    test (count $files) -gt 0; or return
    command grep -lE 'type: "(api|microservice)"' $files 2>/dev/null \
        | command sed -n 's|modules/\([^/]*\)/.*|\1|p' | command sort -u
end

function __talos_sdk_modules
    echo app
    __talos_microservice_modules
end

function __talos_custom_commands
    test -d modules; or return
    set -l files modules/*/src/commands/*Command.ts
    test (count $files) -gt 0; or return
    command grep -h -A1 getName $files 2>/dev/null \
        | command sed -n 's/.*return "\(.*\)".*/\1/p' | command sort -u
end

function __talos_agent_dirs
    printf '%s\n' \
        '.claude'\t'Claude' \
        '.codex'\t'Codex' \
        '.cursor'\t'Cursor' \
        '.gemini'\t'Gemini' \
        '.windsurf'\t'Windsurf' \
        '.cline'\t'Cline' \
        '.junie'\t'JetBrains Junie' \
        '.roo'\t'Roo Code' \
        '.continue'\t'Continue' \
        '.zed'\t'Zed'
end

function __talos_run_commands
    echo install
    set -l pkgfiles
    for f in modules/*/package.json packages/*/package.json
        test -f "$f"; and set -a pkgfiles "$f"
    end
    test (count $pkgfiles) -gt 0; or return
    command awk '
        /"scripts"[[:space:]]*:[[:space:]]*\{/ { inside=1; next }
        inside && /^[[:space:]]*\}/ { inside=0; next }
        inside { if (match($0, /"[^"]+"/)) print substr($0, RSTART+1, RLENGTH-2) }
    ' $pkgfiles 2>/dev/null | command sort -u
end
