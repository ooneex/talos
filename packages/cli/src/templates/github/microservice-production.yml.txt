name: Deploy — Production ({{name}})

on:
  push:
    # Trigger on semver tags, e.g. v1.2.3
    tags: ["v[0-9]+.[0-9]+.[0-9]+"]
  workflow_dispatch:

# Production deploys must never be cancelled mid-flight
concurrency:
  group: deploy-production-{{name}}
  cancel-in-progress: false

permissions:
  contents: read
  deployments: write

jobs:
  ci:
    name: CI
    uses: ./.github/workflows/{{name}}-ci.yml
    with:
      environment: production
    secrets: inherit

  deploy:
    name: Deploy {{name}} to Production
    runs-on: ubuntu-latest
    needs: ci
    # Configure "Required reviewers" on the environment in GitHub Settings
    # to enforce manual approval before this job runs.
    environment:
      name: production-{{name}}
      url: ${{ vars.PRODUCTION_URL_{{NAME_UPPER}} }}
    steps:
      - uses: actions/checkout@v4

      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: latest

      - name: Install dependencies
        run: bun install

      - name: Install Talos CLI
        run: bun add -g @talosjs/cli

      # Restore the accumulating per-version migration cache so already-applied
      # migrations are skipped without opening a database connection. The DB
      # migrations table stays the source of truth; this is only a fast path.
      - name: Cache migrations
        uses: actions/cache@v4
        with:
          path: var/cache/migrations
          key: ${{ runner.os }}-talos-migrations-{{name}}-${{ github.sha }}
          restore-keys: |
            ${{ runner.os }}-talos-migrations-{{name}}-

      # Apply migrations before publishing the release image, so the database is
      # ready by the time the platform pulls the new tag. Expose the production
      # database connection to this step via secrets/vars.
      - name: Run migrations
        run: talos migration:up
        env:
          APP_ENV: production

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Save Docker credentials
        run: >-
          talos docker:credentials:create
          --registry "${{ secrets.DOCKER_REGISTRY || 'docker.io' }}"
          --username "${{ secrets.DOCKER_USERNAME }}"
          --token "${{ secrets.DOCKER_TOKEN }}"
          --silent

      # Builds and pushes only the {{name}} microservice image from
      # modules/{{name}}/Dockerfile, tagged with the release version (e.g.
      # v1.2.3). Deployment is handled by the registry — the platform pulls it.
      - name: Build and publish image
        run: talos docker:publish --modules {{name}} --tag ${{ github.ref_name }} --silent
