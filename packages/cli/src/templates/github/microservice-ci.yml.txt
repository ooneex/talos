name: CI — {{name}}

# Repo-wide lint/test run in the app's ci.yml; this workflow only builds and
# publishes the {{name}} microservice image from modules/{{name}}/Dockerfile.
on:
  push:
    branches: [main, staging, develop]
    paths:
      - "modules/{{name}}/**"
      - "package.json"
      - "bun.lock"
  workflow_call:
    inputs:
      environment:
        description: Target environment
        type: string
        required: false
        default: ci
    outputs:
      image-tag:
        description: Short SHA image tag used for deployment
        value: ${{ jobs.build.outputs.image-tag }}

concurrency:
  group: ci-{{name}}-${{ github.ref }}
  cancel-in-progress: false

permissions:
  contents: read

jobs:
  build:
    name: Build {{name}} Image
    runs-on: ubuntu-latest
    outputs:
      image-tag: sha-${{ steps.sha.outputs.short }}
    permissions:
      contents: read
    steps:
      - uses: actions/checkout@v4

      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: latest

      - name: Install Talos CLI
        run: bun add -g @talosjs/cli

      - name: Compute short SHA
        id: sha
        run: echo "short=$(echo ${{ github.sha }} | cut -c1-7)" >> $GITHUB_OUTPUT

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Save Docker credentials
        run: >-
          talos docker:credentials:create
          --registry "${{ secrets.DOCKER_REGISTRY || 'docker.io' }}"
          --username "${{ secrets.DOCKER_USERNAME }}"
          --token "${{ secrets.DOCKER_TOKEN }}"
          --silent

      # Builds and pushes only the {{name}} microservice image from
      # modules/{{name}}/Dockerfile, tagged with the short commit SHA.
      - name: Build and publish image
        run: talos docker:publish --modules {{name}} --tag sha-${{ steps.sha.outputs.short }} --silent
