name: CI — {{name}}

# Repo-wide lint/test run in the app's ci.yml; this workflow only builds and
# publishes the {{name}} microservice image from modules/{{name}}/Dockerfile.
on:
  push:
    branches: [main, staging, develop]
    paths:
      - "modules/{{name}}/**"
      - "package.json"
      - "bun.lock"
  workflow_call:
    inputs:
      environment:
        description: Target environment
        type: string
        required: false
        default: ci
    outputs:
      image-tag:
        description: Short SHA image tag used for deployment
        value: ${{ jobs.build.outputs.image-tag }}

concurrency:
  group: ci-{{name}}-${{ github.ref }}
  cancel-in-progress: false

permissions:
  contents: read

jobs:
  build:
    name: Build {{name}} Image
    runs-on: ubuntu-latest
    outputs:
      image-tag: sha-${{ steps.sha.outputs.short }}
    permissions:
      contents: read
      packages: write
    steps:
      - uses: actions/checkout@v4

      - name: Compute short SHA
        id: sha
        run: echo "short=$(echo ${{ github.sha }} | cut -c1-7)" >> $GITHUB_OUTPUT

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Log in to GitHub Container Registry
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ghcr.io/${{ github.repository }}/{{name}}
          tags: |
            type=ref,event=branch
            type=sha,prefix=sha-,format=short
            type=raw,value=latest,enable=${{ github.ref == 'refs/heads/main' }}

      - name: Build and push
        uses: docker/build-push-action@v6
        with:
          context: .
          file: modules/{{name}}/Dockerfile
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          # Scope the GitHub Actions cache per microservice so services don't
          # evict each other's layers.
          cache-from: type=gha,scope={{name}}
          cache-to: type=gha,mode=max,scope={{name}}
          platforms: linux/amd64
          build-args: |
            APP_ENV=${{ inputs.environment || 'ci' }}
            APP_VERSION=${{ github.sha }}
