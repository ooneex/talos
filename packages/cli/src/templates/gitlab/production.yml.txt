# GitLab CI — Production Deployment
# Place at .gitlab/ci/production.yml and include from .gitlab-ci.yml:
#
#   include:
#     - local: .gitlab/ci/ci.yml
#     - local: .gitlab/ci/production.yml
#
# Required CI/CD variables (Settings → CI/CD → Variables):
#   DOCKER_REGISTRY  — Container registry host (defaults to docker.io)
#   DOCKER_USERNAME  — Registry username
#   DOCKER_TOKEN     — Registry access token (mark as "Masked")
#   PRODUCTION_URL   — App URL shown on the GitLab environment (e.g. https://app.example.com)
#   (plus the production database connection consumed by `talos migration:up`)
#
# Protect the production environment in Settings → CI/CD → Environments and add
# required approvers to enforce manual sign-off before deploy.

stages:
  - quality
  - build
  - deploy

deploy-production:
  stage: deploy
  image: docker:27
  services:
    - docker:27-dind
  needs:
    - quality
  variables:
    DOCKER_DRIVER: overlay2
    DOCKER_TLS_CERTDIR: /certs
  environment:
    name: production
    url: $PRODUCTION_URL
  # Install Bun and the Talos CLI into the Docker image so migrations run and
  # `talos docker:publish` can build and push. Bun's global bin lands in ~/.bun/bin.
  before_script:
    - apk add --no-cache bash curl unzip
    - curl -fsSL https://bun.sh/install | bash
    - export PATH="$HOME/.bun/bin:$PATH"
    - bun add -g @talosjs/cli
  script:
    # Apply migrations before publishing the release image, so the database is
    # ready by the time the platform pulls the new tag. Expose the production
    # database connection to this job via CI/CD variables.
    - APP_ENV=production talos migration:up
    - >-
      talos docker:credentials:create
      --registry "${DOCKER_REGISTRY:-docker.io}"
      --username "$DOCKER_USERNAME"
      --token "$DOCKER_TOKEN"
      --silent
    # Builds and pushes an image for every package/module that ships a
    # Dockerfile, tagged with the release version (e.g. v1.2.3). Deployment is
    # handled by the registry — the platform pulls the new tag.
    - talos docker:publish --tag "$CI_COMMIT_REF_NAME" --silent
  rules:
    # Auto-deploy on semver tags
    - if: $CI_COMMIT_TAG =~ /^v[0-9]+\.[0-9]+\.[0-9]+$/
    # Manual trigger from main branch
    - if: $CI_COMMIT_BRANCH == "main"
      when: manual
      allow_failure: false
