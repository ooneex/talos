# GitLab CI — Production Deployment
# Place at .gitlab/ci/production.yml and include from .gitlab-ci.yml:
#
#   include:
#     - local: .gitlab/ci/ci.yml
#     - local: .gitlab/ci/production.yml
#
# Required CI/CD variables (Settings → CI/CD → Variables):
#   PROD_HOST             — Production server hostname or IP
#   PROD_USER             — SSH username
#   PROD_SSH_KEY          — SSH private key (mark as "File" type)
#   PROD_SSH_KNOWN_HOSTS  — Output of: ssh-keyscan <PROD_HOST>
#   PRODUCTION_URL        — App URL used for health checks (e.g. https://app.example.com)
#
# Protect the production environment in Settings → CI/CD → Environments
# and add required approvers to enforce manual sign-off before deploy.

stages:
  - quality
  - build
  - deploy

deploy-production:
  stage: deploy
  image: alpine:latest
  needs:
    - job: build-image
      artifacts: true
  environment:
    name: production
    url: $PRODUCTION_URL
  before_script:
    - apk add --no-cache openssh-client curl
    - eval $(ssh-agent -s)
    - echo "$PROD_SSH_KEY" | tr -d '\r' | ssh-add -
    - mkdir -p ~/.ssh && chmod 700 ~/.ssh
    - echo "$PROD_SSH_KNOWN_HOSTS" >> ~/.ssh/known_hosts
    - chmod 600 ~/.ssh/known_hosts
  script:
    - IMAGE=$CI_REGISTRY_IMAGE:$IMAGE_TAG
    - |
      ssh $PROD_USER@$PROD_HOST \
        CI_REGISTRY="$CI_REGISTRY" \
        CI_REGISTRY_USER="$CI_REGISTRY_USER" \
        CI_REGISTRY_PASSWORD="$CI_REGISTRY_PASSWORD" \
        CI_REGISTRY_IMAGE="$CI_REGISTRY_IMAGE" \
        IMAGE="$IMAGE" \
        'bash -s' << 'ENDSSH'
          set -e

          echo "$CI_REGISTRY_PASSWORD" | docker login "$CI_REGISTRY" -u "$CI_REGISTRY_USER" --password-stdin

          # Preserve current image as rollback target before pulling new one
          docker tag "$CI_REGISTRY_IMAGE:latest" "$CI_REGISTRY_IMAGE:rollback" 2>/dev/null || true

          docker pull "$IMAGE"

          cd /opt/{{NAME}}

          # Run migrations before switching traffic to catch failures early
          docker run --rm --env-file .env "$IMAGE" bun run db:migrate

          APP_IMAGE="$IMAGE" docker compose up -d --no-deps --wait app
          docker image prune -f --filter "until=24h"
      ENDSSH
    - |
      echo "Verifying production health at $PRODUCTION_URL..."
      for i in $(seq 1 24); do
        STATUS=$(curl -sf -o /dev/null -w "%{http_code}" "$PRODUCTION_URL/health" 2>/dev/null || echo "000")
        [ "$STATUS" = "200" ] && echo "Production is healthy (attempt $i)" && exit 0
        echo "Attempt $i/24 — HTTP $STATUS, retrying in 5s..."
        sleep 5
      done
      echo "Health check failed — rolling back to previous image"
      ssh $PROD_USER@$PROD_HOST \
        CI_REGISTRY_IMAGE="$CI_REGISTRY_IMAGE" \
        'bash -s' << 'ENDSSH'
          cd /opt/{{NAME}}
          APP_IMAGE="$CI_REGISTRY_IMAGE:rollback" docker compose up -d --no-deps app
      ENDSSH
      exit 1
  rules:
    # Auto-deploy on semver tags
    - if: $CI_COMMIT_TAG =~ /^v[0-9]+\.[0-9]+\.[0-9]+$/
    # Manual trigger from main branch
    - if: $CI_COMMIT_BRANCH == "main"
      when: manual
      allow_failure: false
