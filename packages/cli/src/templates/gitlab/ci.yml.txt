# GitLab CI — Core CI Pipeline
# Place at .gitlab/ci/ci.yml and include from .gitlab-ci.yml:
#
#   include:
#     - local: .gitlab/ci/ci.yml
#
# Required CI/CD variables (Settings → CI/CD → Variables):
#   DOCKER_REGISTRY  — Container registry host (defaults to docker.io)
#   DOCKER_USERNAME  — Registry username
#   DOCKER_TOKEN     — Registry access token (mark as "Masked")

stages:
  - quality
  - build

quality:
  image: oven/bun:latest
  stage: quality
  variables:
    BUN_INSTALL_CACHE_DIR: $CI_PROJECT_DIR/.bun-cache
  cache:
    key:
      files:
        - bun.lock
    paths:
      - .bun-cache/
      - var/cache/monorepo/
    policy: pull-push
  script:
    - bun add -g @talosjs/cli
    - talos monorepo:check
  rules:
    - if: $CI_PIPELINE_SOURCE == "merge_request_event"
    - if: $CI_COMMIT_BRANCH =~ /^(main|staging|develop)$/
    - if: $CI_COMMIT_TAG

build-image:
  stage: build
  image: docker:27
  services:
    - docker:27-dind
  needs:
    - quality
  variables:
    DOCKER_DRIVER: overlay2
    DOCKER_TLS_CERTDIR: /certs
  # Only build images on branch commits (not merge requests)
  rules:
    - if: $CI_COMMIT_BRANCH =~ /^(main|staging|develop)$/
  # Install Bun and the Talos CLI into the Docker image so `talos docker:publish`
  # can build and push. Bun's global bin (with `talos`) lands in ~/.bun/bin.
  before_script:
    - apk add --no-cache bash curl unzip
    - curl -fsSL https://bun.sh/install | bash
    - export PATH="$HOME/.bun/bin:$PATH"
    - bun add -g @talosjs/cli
  script:
    - SHORT_SHA=$(echo "$CI_COMMIT_SHA" | cut -c1-7)
    - >-
      talos docker:credentials:create
      --registry "${DOCKER_REGISTRY:-docker.io}"
      --username "$DOCKER_USERNAME"
      --token "$DOCKER_TOKEN"
      --silent
    # Builds and pushes an image for every package/module that ships a
    # Dockerfile, tagging each with the short commit SHA.
    - talos docker:publish --tag "sha-$SHORT_SHA" --silent
