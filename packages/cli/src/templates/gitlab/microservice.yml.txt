# GitLab CI — {{name}} microservice
# Include from .gitlab-ci.yml:
#
#   include:
#     - local: .gitlab/ci/{{name}}.yml
#
# Builds and deploys the {{name}} microservice image from modules/{{name}}/Dockerfile.
# Repo-wide lint/test run in the core ci.yml pipeline, so they are not repeated here.
#
# Required CI/CD variable (in addition to the app's):
#   PRODUCTION_URL_{{NAME_UPPER}}  — {{name}} environment URL shown in GitLab

{{name}}-build-image:
  stage: build
  image: docker:27
  services:
    - docker:27-dind
  variables:
    DOCKER_DRIVER: overlay2
    DOCKER_TLS_CERTDIR: /certs
  # Only build on branch commits that touch this microservice
  rules:
    - if: $CI_COMMIT_BRANCH =~ /^(main|staging|develop)$/
      changes:
        - modules/{{name}}/**/*
  # Install Bun and the Talos CLI into the Docker image so `talos docker:publish`
  # can build and push. Bun's global bin (with `talos`) lands in ~/.bun/bin.
  before_script:
    - apk add --no-cache bash curl unzip
    - curl -fsSL https://bun.sh/install | bash
    - export PATH="$HOME/.bun/bin:$PATH"
    - bun add -g @talosjs/cli
  script:
    - SHORT_SHA=$(echo "$CI_COMMIT_SHA" | cut -c1-7)
    - >-
      talos docker:credentials:create
      --registry "${DOCKER_REGISTRY:-docker.io}"
      --username "$DOCKER_USERNAME"
      --token "$DOCKER_TOKEN"
      --silent
    # Builds and pushes only the {{name}} microservice image from
    # modules/{{name}}/Dockerfile, tagged with the short commit SHA.
    - talos docker:publish --modules {{name}} --tag "sha-$SHORT_SHA" --silent

{{name}}-deploy-production:
  stage: deploy
  image: docker:27
  services:
    - docker:27-dind
  variables:
    DOCKER_DRIVER: overlay2
    DOCKER_TLS_CERTDIR: /certs
  environment:
    name: production-{{name}}
    url: $PRODUCTION_URL_{{NAME_UPPER}}
  before_script:
    - apk add --no-cache bash curl unzip
    - curl -fsSL https://bun.sh/install | bash
    - export PATH="$HOME/.bun/bin:$PATH"
    - bun add -g @talosjs/cli
  script:
    # Apply migrations before publishing the release image, so the database is
    # ready by the time the platform pulls the new tag.
    - APP_ENV=production talos migration:up
    - >-
      talos docker:credentials:create
      --registry "${DOCKER_REGISTRY:-docker.io}"
      --username "$DOCKER_USERNAME"
      --token "$DOCKER_TOKEN"
      --silent
    # Builds and pushes only the {{name}} microservice image from
    # modules/{{name}}/Dockerfile, tagged with the release version (e.g. v1.2.3).
    - talos docker:publish --modules {{name}} --tag "$CI_COMMIT_REF_NAME" --silent
  rules:
    # Auto-deploy on semver tags
    - if: $CI_COMMIT_TAG =~ /^v[0-9]+\.[0-9]+\.[0-9]+$/
    # Manual trigger from main branch
    - if: $CI_COMMIT_BRANCH == "main"
      when: manual
      allow_failure: false
